### 1.1 整体架构

#### 核心代码（`agent.py`）

```python
async def get_agent_stream_response(
        query: str,
        session_id: str,
        user_id: str,
        custom_tools: Optional[List[BaseTool]] = None,
        **kwargs
) -> AsyncGenerator[str, None]:
    # 1. 从工厂获取全新的Executor实例
    agent_executor = agent_factory.create_agent_executor(
        custom_tools=custom_tools, **kwargs
    )
    
    # 2. 构建聊天历史
    chat_history: List[BaseMessage] = []
    if history:
        for user_msg, assistant_msg in history:
            chat_history.append(HumanMessage(content=user_msg))
            chat_history.append(AIMessage(content=assistant_msg))
    
    # 3. 流式执行Agent
    async for chunk in agent_executor.astream({
        "input": query,
        "chat_history": chat_history,
        "system_prompt": agent_factory.default_system_prompt
    }):
        if "output" in chunk:
            # 输出最终答案
            yield f"data: {json.dumps({'type': 'response', 'content': chunk['output']}, ensure_ascii=False)}\n\n"
        elif "intermediate_steps" in chunk:
            for action, observation in chunk["intermediate_steps"]:
                # 记录思考过程和工具调用
                steps.append({
                    "thought": action.log,
                    "tool": action.tool,
                    "tool_input": action.tool_input,
                    "tool_output": observation
                })
```

### 1.3 AgentFactory工厂模式

```python
class AgentFactory:
    def __init__(
            self,
            model: str = "qwen3-max",
            api_key: Optional[str] = None,
            default_tools: Optional[List[BaseTool]] = None,
            default_middleware: Optional[List] = None,
            default_system_prompt: Optional[str] = None,
    ):
        self.model = model
        self.api_key = api_key or os.getenv("CHAT_API_KEY")
        self.default_tools = default_tools or self._get_default_tools()
        self.default_middleware = default_middleware or self._get_default_middleware()
        self.default_system_prompt = default_system_prompt or self._get_default_system_prompt()
    
    def create_agent_executor(
            self,
            custom_tools: Optional[List[BaseTool]] = None,
            custom_model: Optional[str] = None,
            custom_system_prompt: Optional[str] = None,
            verbose: bool = True,
            return_intermediate_steps: bool = True,
            **kwargs
    ) -> AgentExecutor:
        # 每次创建新实例，避免状态污染
        chat_model = self._create_chat_model(custom_model)
        prompt = self._create_prompt()
        tools = custom_tools or self.default_tools
        
        agent = create_tool_calling_agent(chat_model, tools, prompt)
        return AgentExecutor(
            agent=agent,
            tools=tools,
            verbose=verbose,
            return_intermediate_steps=return_intermediate_steps,
            **kwargs
        )
```

### 2.1 Hybrid Retriever混合检索

#### 核心实现（`hybrid_retriever.py`）

```python
class HybridRetriever:
    def __init__(self, vectors_store: Chroma):
        self.vectors_store = vectors_store
    
    async def get_bm25_retriever(self, user_id: str = None):
        # 获取用户的所有文档，构建BM25索引
        all_docs_result = await asyncio.to_thread(
            self.vectors_store.get,
            include=['documents', 'metadatas'],
            where={'user_id': user_id}
        )
        documents = []
        for i, doc_content in enumerate(all_docs_result['documents']):
            metadata = all_docs_result['metadatas'][i]
            documents.append(Document(page_content=doc_content, metadata=metadata))
        
        if documents:
            return BM25Retriever.from_documents(
                documents=documents,
                k=chroma_config['k']
            )
        return None
    
    async def get_retriever(self, query: str = None, user_id: str = None) -> BaseRetriever:
        if not user_id:
            return EmptyRetriever()
        
        # 向量检索器
        filter_dict = {'user_id': user_id}
        vector_retriever = self.vectors_store.as_retriever(
            search_type='similarity',
            search_kwargs={'k': chroma_config['k'], 'filter': filter_dict},
        )
        
        # BM25检索器
        bm25_retriever = await self.get_bm25_retriever(user_id)
        
        if bm25_retriever:
            # 动态调整权重
            weights = await self.get_dynamic_weights(query)
            ensemble_retriever = EnsembleRetriever(
                retrievers=[vector_retriever, bm25_retriever],
                weights=weights
            )
            return ensemble_retriever
        else:
            return vector_retriever
    
    @staticmethod
    async def get_dynamic_weights(query: str = None):
        default_vector_weight = 0.5
        default_bm25_weight = 0.5
        
        if not query:
            return [default_vector_weight, default_bm25_weight]
        
        query_length = len(query)
        query_words = len(query.split())
        
        # 长查询倾向向量检索
        if query_length > 50:
            vector_weight = 0.7
            bm25_weight = 0.3
        # 短查询倾向BM25关键词检索
        elif query_length < 20:
            vector_weight = 0.3
            bm25_weight = 0.7
        else:
            vector_weight = default_vector_weight
            bm25_weight = default_bm25_weight
        
        return [vector_weight, bm25_weight]
```

class RagService:
    async def get_documents_and_summary(self, query: str) -> dict:
        # 1. 检索文档
        documents = await self.retrieve_document(query)
        
        # 2. 提取文档内容
        document_contents = [doc.page_content for doc in documents]
        
        # 3. 重排序
        reordered_documents = await self.reorder_documents(query, document_contents)
        
        # 4. 分批总结（并发处理）
        individual_summaries = []
        tasks = []
        for i, doc in enumerate(reordered_documents[:3]):
            tasks.append(self._summarize_single_doc(i, doc, query))
        
        individual_summaries = await asyncio.gather(*tasks)
        
        # 5. 合并总结
        combined_context = "以下是多个文档的摘要:\n"
        for i, summary in enumerate(individual_summaries, 1):
            combined_context += f"【文档{i}摘要】: {summary}\n\n"
        
        # 6. 生成最终答案
        final_summary = await self.chain.ainvoke({
            "input": query,
            "context": combined_context
        })
        
        return {
            "documents": reordered_documents,
            "summary": final_summary
        }

## 三、HyDE技术实现方案

### 3.1 Hypothetical Document生成

```python
class RagService:
    def __init__(self, user_id: str = None):
        self.hyde_prompt_template = PromptTemplate.from_template(
            "基于以下问题，生成一个详细的假设性回答，"
            "我将根据你的这个假设性回答在向量数据库里检索文档：\n\n"
            "问题：{query}\n\n"
            "假设性回答："
        )
    
    async def generate_hypothetical_document(self, query: str) -> str:
        try:
            hyde_chain = (
                self.hyde_prompt_template
                | self.chat_model
                | StrOutputParser()
            )
            hypothetical_doc = await hyde_chain.ainvoke({"query": query})
            logger.info(f"【HyDE】生成的假设性文档:\n{hypothetical_doc}")
            return hypothetical_doc
        except Exception as e:
            logger.error(f"【HyDE】生成假设性文档失败: {e}")
            return query  # 回退到原始查询
```

### 3.2 HyDE检索流程

```python
async def retrieve_document(self, query: str) -> list:
    if not self.user_id:
        return []
    
    try:
        # 1. 初始化检索器
        if self.retriever is None:
            await self.initialize_retriever(query)
        
        # 2. 生成假设性文档
        logger.info(f"【HyDE】开始处理查询: {query}")
        hypothetical_doc = await self.generate_hypothetical_document(query)
        
        # 3. 使用假设性文档进行检索
        logger.info(f"【HyDE】使用假设性文档进行检索")
        documents = await self.retriever.ainvoke(hypothetical_doc)
        logger.info(f"【HyDE】检索到 {len(documents)} 个相关文档")
        
        return documents
    except Exception as e:
        logger.error(f"【HyDE】检索文档失败: {e}")
        return []
```

### 3.3 HyDE技术优势

1. **解决语义鸿沟**：将用户查询转化为更接近文档风格的回答
2. **提高召回率**：假设性回答包含更多与目标文档匹配的关键词
3. **冷启动友好**：即使相关文档较少，也能提高匹配概率

## 四、缓存体系实现

### 4.1 Redis缓存架构

```python
class RedisCache(Generic[T]):
    @staticmethod
    async def get_or_set(
            key: str,
            func: Callable[..., T],
            *args,
            expire: int = 3600,
            **kwargs
    ) -> T:
        # 1. 尝试获取缓存
        cache_key = str(key)
        cached_data = await get_redis_cache_json(cache_key)
        if cached_data is None:
            cached_data = await get_redis_cache_str(cache_key)
        
        if cached_data is not None:
            return cached_data
        
        # 2. 执行函数
        result = await func(*args, **kwargs)
        
        # 3. 转换为可序列化格式
        serializable_result = convert_to_serializable(result)
        
        # 4. 设置缓存
        await set_redis_cache(cache_key, serializable_result, expire)
        
        return result
```

### 4.2 缓存装饰器

```python
def cache_with_redis(prefix: str, expire: int = 3600):
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        async def wrapper(*args, **kwargs):
            key = RedisCache.cache_key(prefix, *args, **kwargs)
            return await RedisCache.get_or_set(key, func, *args, expire=expire, **kwargs)
        return wrapper
    return decorator
```

### 4.3 缓存键生成

```python
@staticmethod
def cache_key(prefix: str, *args, **kwargs) -> str:
    parts = [prefix]
    
    # 添加位置参数（排除数据库会话）
    for arg in args:
        if arg is not None and not hasattr(arg, 'execute'):
            parts.append(str(arg))
    
    # 添加关键字参数
    for key, value in sorted(kwargs.items()):
        if value is not None and key != 'db':
            parts.append(f"{key}:{value}")
    
    return ":".join(parts)
```

### 5.1 MD5哈希管理

```python
class MD5Store:
    def _get_md5_store_dir(self, user_id: str = None) -> str:
        if user_id:
            return os.path.join(self.base_dir, 'user_md5', user_id)
        else:
            return os.path.join(self.base_dir, 'public_md5')
    
    async def check_md5_hex(self, md5_for_check: str, user_id: str = None) -> bool:
        md5_dir = self._get_md5_store_dir(user_id)
        md5_path = os.path.join(md5_dir, 'md5_hex_store.txt')
        
        if not await aio_os.path.exists(md5_dir):
            return False
        
        try:
            async with aiofiles.open(md5_path, 'r', encoding="utf-8") as f:
                async for line in f:
                    line = line.strip()
                    if line.startswith('{'):
                        data = json.loads(line)
                        if data.get('md5') == md5_for_check:
                            return True
                    else:
                        if line == md5_for_check:
                            return True
            return False
        except Exception as e:
            logger.error(f"【向量数据库】检查MD5时出错: {e}")
            return False
```

### 6.2 Agent流式响应生成器

```python
async def get_agent_stream_response(
        query: str,
        session_id: str,
        user_id: str,
        custom_tools: Optional[List[BaseTool]] = None,
        **kwargs
) -> AsyncGenerator[str, None]:
    try:
        logger.info(f"【Agent流式响应】开始处理请求，用户ID: {user_id}")
        
        set_current_user_id(user_id)
        
        # 获取会话历史
        history = await sm.session_manager.get_history(session_id, user_id)
        
        # 构建聊天历史
        chat_history: List[BaseMessage] = []
        if history:
            from langchain_core.messages import HumanMessage, AIMessage
            for user_msg, assistant_msg in history:
                chat_history.append(HumanMessage(content=user_msg))
                chat_history.append(AIMessage(content=assistant_msg))
        
        # 从工厂获取全新的Executor实例
        agent_executor = agent_factory.create_agent_executor(
            custom_tools=custom_tools, **kwargs
        )
        
        # 先发送初始响应
        yield f"data: {json.dumps({'type': 'response', 'content': '', 'session_id': session_id}, ensure_ascii=False)}\n\n"
        
        # 使用agent_executor的astream方法获取流式响应
        full_response = []
        async for chunk in agent_executor.astream({
            "input": query,
            "chat_history": chat_history,
            "system_prompt": agent_factory.default_system_prompt
        }):
            if "output" in chunk:
                chunk_content = chunk["output"]
                full_response.append(chunk_content)
                
                # 逐字发送输出，实现流式效果
                for char in chunk_content:
                    yield f"data: {json.dumps({'type': 'response', 'content': char}, ensure_ascii=False)}\n\n"
                    await asyncio.sleep(0.02)
        
        # 添加到会话历史
        await sm.session_manager.add_message(
            session_id, user_id, query, ''.join(full_response)
        )
        
        # 发送结束标记
        yield f"data: {json.dumps({'type': 'done', 'session_id': session_id}, ensure_ascii=False)}\n\n"
    
    except Exception as e:
        logger.error(f"【Agent流式响应】处理请求失败: {e}", exc_info=True)
        error_message = f"错误: {str(e)}"
        yield f"data: {json.dumps({'type': 'error', 'content': error_message, 'session_id': session_id}, ensure_ascii=False)}\n\n"
        yield f"data: {json.dumps({'type': 'done'}, ensure_ascii=False)}\n\n"
```

### 6.3 SSE事件类型

| 类型 | 说明 | 数据内容 |
|------|------|----------|
| `response` | 正常响应内容 | `content` - 响应文本片段 |
| `done` | 响应结束 | `session_id` - 会话ID |
| `error` | 错误信息 | `content` - 错误描述 |

## 总结

本项目通过以下六大技术实现了一个完整的Agent智能问答系统：

1. **ReAct Agent推理引擎** - 基于LangChain的工具调用机制，支持多轮推理
2. **Hybrid RAG检索** - BM25 + 向量检索的混合检索，动态权重调整
3. **HyDE技术** - 假设性文档生成，提高检索召回率
4. **Redis缓存体系** - 会话、文档元数据、限流计数的多级缓存
5. **知识库隔离** - 基于用户ID和MD5哈希的文档权限控制
6. **SSE流式推送** - 实时响应、逐字输出的良好用户体验

整个系统采用异步架构设计，使用FastAPI作为Web框架，Chroma作为向量数据库，Redis作为缓存，实现了高性能、可扩展的智能问答服务。
